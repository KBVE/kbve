import { Aside } from '@astrojs/starlight/components';
import BentoGrid from '@/components/bento/BentoGrid.astro';

export const contentModules = import.meta.glob('./_content/*.mdx', { eager: true });

export const bentoItems = Object.entries(contentModules).map(([path, mod]) => {
    const frontmatter = mod.frontmatter || {};
    return {
        bentoId: frontmatter.bentoId || path.split('/').pop().replace('.mdx', ''),
        title: frontmatter.title || path.split('/').pop().replace('.mdx', ''),
        description: frontmatter.description || '',
        bentoSize: frontmatter.bentoSize || 'medium',
        bentoOrder: frontmatter.bentoOrder ?? 99,
        bentoColor: frontmatter.bentoColor || 'from-zinc-700/30 to-zinc-900/60',
        icon: frontmatter.icon || 'file-text',
        Content: mod.Content,
    };
});
